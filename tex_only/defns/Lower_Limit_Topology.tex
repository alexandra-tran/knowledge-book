\begin{defn*}{Lower Limit Topology}
    \begin{center}
         The topology generated by $ \mathcal{ B } =  \left\{ \left[ a, b \right) \subseteq \mathbb{R}: a < b \right\}  $ is defined as the lower limit topology on $ R $
    \end{center}
    \begin{rmk}
        \begin{center}
            The corresponding topological space $ \left( \mathbb{R} , \mathcal{ S }  \right) $ is called the Sorgenfrey line.
        \end{center}
    \end{rmk}
\end{defn*}
