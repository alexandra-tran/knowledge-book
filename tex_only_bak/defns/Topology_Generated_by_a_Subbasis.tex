\begin{defn*}{Topology Generated by a Subbasis}
    The topology generated by the a subbasis $ \mathcal{ S }  $ is defined to be the collection $ \mathcal{ T } _{ \mathcal{ S }  }  $ of all unions of finite intersections of $ \mathcal{ S }  $.
\end{defn*}
