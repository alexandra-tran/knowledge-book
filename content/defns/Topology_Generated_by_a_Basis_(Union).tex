\begin{definition}
{Topology Generated by a Basis (Union)}
    Let \(X\) be a set and \(\mathcal{B}\) a basis on \(X\), we define:
    \[
    \mathcal{T}_{\mathcal{B}} = \left\{ \bigcup \mathcal{C}: \mathcal{C} \subseteq \mathcal{B}  \right\}
    \]
    and say that $ \mathcal{T} _{B}$ is called the topology generated by $ \mathcal{B}$       
\end{definition}
