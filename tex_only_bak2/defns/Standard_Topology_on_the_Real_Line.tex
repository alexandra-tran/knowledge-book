\begin{defn*}{Standard Topology on the Real Line}
    If $ \mathcal{ B }  $ is the collection of all open intervals on the real line, that is: 
    \[
    \left( a, b \right) = \left\{ x : a < x < b \right\} 
    \]
    then the topology generated by $ \mathcal{ B }  $ is called the standard topology on the real line
\end{defn*}
